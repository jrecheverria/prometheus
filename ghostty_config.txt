# Appearance
font-family = JetBrains Mono
font-size = 14
theme = Arthur
background-opacity = 0.95

# Spacing
window-padding-x = 16
window-padding-y = 16
window-padding-balance = true

# Cursor
cursor-style = bar
cursor-style-blink = false

# Window chrome
macos-titlebar-style = tabs
window-save-state = always

# Scrollback
scrollback-limit = 100000

# Mouse
mouse-hide-while-typing = true

# Shell integration niceties
shell-integration-features = cursor,sudo,title

# Quick terminal (drop-down style, bind to a hotkey)
keybind = global:cmd+grave_accent=toggle_quick_terminal

# Hide the top bar in ghostty for a bezel-less look
macos-titlebar-style = hidden

# --- Splits ---
keybind = cmd+d=new_split:right
keybind = cmd+shift+d=new_split:down

# Move focus between splits (vim-style hjkl)
keybind = cmd+h=goto_split:left
keybind = cmd+j=goto_split:bottom
keybind = cmd+k=goto_split:top
keybind = cmd+l=goto_split:right

# Resize splits (shift + hjkl)
keybind = cmd+shift+h=resize_split:left,20
keybind = cmd+shift+j=resize_split:down,20
keybind = cmd+shift+k=resize_split:up,20
keybind = cmd+shift+l=resize_split:right,20

# Equalize and zoom
keybind = cmd+e=equalize_splits
keybind = cmd+z=toggle_split_zoom

# --- Tabs ---
keybind = cmd+t=new_tab
keybind = cmd+w=close_surface
keybind = cmd+n=next_tab
keybind = cmd+p=previous_tab

# Jump to specific tabs
keybind = cmd+1=goto_tab:1
keybind = cmd+2=goto_tab:2
keybind = cmd+3=goto_tab:3
keybind = cmd+4=goto_tab:4
keybind = cmd+5=goto_tab:5

# --- Windows ---
keybind = cmd+shift+n=new_window
